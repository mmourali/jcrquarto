\renewcommand{\underline}[1]{#1}

% ---- MANUSCRIPT SETUP ----
\newcommand{\jcrsetupmanuscript}{%
  \pagestyle{empty}%
  \doublespacing%
  \raggedright%
  \setlength{\parindent}{0.5in}%
  \titleformat{\section}%
    {\normalfont\normalsize\bfseries\centering\MakeUppercase}{}{0pt}{}%
  \titlespacing*{\section}{0pt}{\baselineskip}{\baselineskip}%
  \titleformat{\subsection}%
    {\normalfont\normalsize}{}{0pt}{}%
  \titlespacing*{\subsection}{0pt}{\baselineskip}{\baselineskip}%
  \titleformat{\subsubsection}[runin]%
    {\normalfont\normalsize\itshape}{}{0pt}{}[.]%
  \titlespacing*{\subsubsection}{\parindent}{\baselineskip}{0.5em}%
}

% ---- ARTICLE SETUP ----
\newcommand{\jcrsetuparticle}{%
  % Centered page numbers at bottom, no headers
  \pagestyle{fancy}%
  \fancyhf{}%
  \fancyfoot[C]{\small\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
  \fancypagestyle{plain}{%
    \fancyhf{}%
    \fancyfoot[C]{\small\thepage}%
    \renewcommand{\headrulewidth}{0pt}%
  }%
  \thispagestyle{plain}%
  \setlength{\parindent}{1.5em}%
  % Headings: bold italic, left-aligned (matches amaquarto article style)
  \titleformat{\section}%
    {\normalfont\large\bfseries\itshape\raggedright}{}{0pt}{}%
  \titlespacing*{\section}{0pt}{18pt}{6pt}%
  \titleformat{\subsection}%
    {\normalfont\normalsize\bfseries\itshape\raggedright}{}{0pt}{}%
  \titlespacing*{\subsection}{0pt}{14pt}{4pt}%
  \titleformat{\subsubsection}%
    {\normalfont\normalsize\itshape\raggedright}{}{0pt}{}%
  \titlespacing*{\subsubsection}{0pt}{10pt}{3pt}%
  \captionsetup{font=small,labelfont={bf},textfont={}}%
}
